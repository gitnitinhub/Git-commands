//git commands for easy use 

L-1
>>Cofiguration of git for very first time after installing it 

git --version						{to check the git version installed on the device}
git config --global user.name "gitnitinhub" 		{to set user name for the very first time}
git config --global user.email "exapmple@email.com" 	{to set email id for the very first time}
git config --global core.editor "code --wait" 		{to set the default text editor for git usage}
git config --global init.defaultbranch main		{to set default branch as main}
git config --list					{to see all the command passed above in a list}
git config user.name					{to see the set user name for git }
git config user.email					{to see current email address for git acc}
